create table price(
product_id int,
start_date date,
end_date date,
price INT
);

CREATE TABLE UnitsSold (
    product_id INT,
    purchase_date DATE,
    units INT
);

INSERT INTO UnitsSold VALUES
(1,'2019-02-25',100),
(1,'2019-03-01',15),
(2,'2019-02-10',200),
(2,'2019-03-22',30);
# Write your MySQL query statement below
SELECT 
p.product_id,
ROUND(
    coalesce(SUM(p.price * u.units) / SUM(u.units),0),
    2
) AS average_price
FROM Price p
left JOIN UnitsSold u
ON p.product_id = u.product_id
AND u.purchase_date BETWEEN p.start_date AND p.end_date
GROUP BY p.product_id;
